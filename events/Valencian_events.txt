# Sindicat Remença
country_event = {
	id = flavor_ara.6
	title = "flavor_ara.EVTNAME6"
	desc = "flavor_ara.EVTDESC6"
	picture = ELECTION_REPUBLICAN_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = VAL
		NOT = { is_year = 1500 }
		OR = { 
			adm = 4 
			dip = 4 
		}
		any_owned_province = {
			culture = catalan
		}
		NOT = { 
			any_owned_province = {
				has_province_modifier = ara_sindicat_remenca 
			}
		}
	}
	
	mean_time_to_happen = {
		months = 200
	}
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				limit = {
					is_core = ROOT
					culture = catalan
				}
				set_province_flag = ara_sindicat_remença_flag
			}
			random_owned_province = {
				limit = {
					is_core = ROOT
					culture = catalan
					NOT = {  has_province_flag = ara_sindicat_remença_flag }
				}
				set_province_flag = ara_sindicat_remença_flag
			}
			random_owned_province = {
				limit = {
					is_core = ROOT
					culture = catalan
					NOT = { has_province_flag = ara_sindicat_remença_flag }
				}
				set_province_flag = ara_sindicat_remença_flag
			}
		}
	}
	
	option = {
		name = "flavor_ara.EVTOPTA6"
		add_country_modifier = {
			name = "decentralization_modifier"
			duration = 3650
		}
		every_owned_province = {
			limit = {
				has_province_flag = ara_sindicat_remença_flag
			}
			add_local_autonomy = 25
			add_province_modifier = { 
				name = ara_sindicat_remenca 
				duration = -1
			}
		}
		hidden_effect = {
			every_owned_province = {
				limit = {
					has_province_flag = ara_sindicat_remença_flag
				}
				clr_province_flag = ara_sindicat_remença_flag
			}
		}
	}
	option = {
		name = "flavor_ara.EVTOPTB6"
		every_owned_province = {
			limit = {
				has_province_flag = ara_sindicat_remença_flag
			}
			clr_province_flag = ara_sindicat_remença_flag
			spawn_rebels = {
				type = anti_tax_rebels
				size = 1
			}
		}
	}
}
